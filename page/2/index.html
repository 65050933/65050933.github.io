//路径压缩
int find(int x){
return p[x] == x ? p[x] : p[x] = find(p[x]);
}
//按秩合并
void merge(int x,int y){
int rx = find(x);
int ry = find(y);
if(rx != ry){
if(rank[rx] <= rank[ry]){
p[rx] = ry;
if(rank[rx] == rank[ry]){
rank[ry]++;
}
}
else{
p[ry] = rx;
}
}
}
